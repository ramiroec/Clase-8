Manipulación de Cadenas en JavaScript

Convertir y Redondear

Convierte una cadena en un número y lo redondea a dos decimales.

function convertirYRedondear(cadena) {
  let numero = Number(cadena);
  if (isNaN(numero)) {
    return "La cadena no es un número válido.";
  }
  let redondeado = numero.toFixed(2);
  return redondeado;
}

Extraer Entero

Extrae el número entero de una cadena.

function extraerEntero(cadena) {
  let entero = parseInt(cadena, 10);
  if (isNaN(entero)) {
    return "La cadena no contiene un número entero válido.";
  }
  return entero;
}
